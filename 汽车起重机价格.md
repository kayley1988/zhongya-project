document.addEventListener('mouseup', function() {

  if (selectedText) {
    // 右键菜单添加查询选项
    GM_registerMenuCommand('查询工程机械信息', function() {
      searchEquipment(selectedText);
    });
  }

// 2. 多平台数据抓取
function searchEquipment(keyword) {
  // 本地缓存优先
  const cachedData = GM_getValue(keyword);
  if (cachedData) {
    showResult(cachedData);
    return;
  }

  // 多线程并发获取数据
  Promise.all([
    getParamsFromLmjx(keyword),   // 中国路面机械网参数
    getPriceFromXcmg(keyword),    // 徐工官网价格
    getUsedPriceFromTieba(keyword),// 铁甲二手机价格
    getRentPriceFromTieba(keyword)// 铁甲租赁价格
  ]).then(showResult);
}

// 3. 结果展示
function showResult(data) {
  // 创建悬浮弹窗
  const popup = document.createElement('div');
  popup.id = 'machinery-info-popup';
  // 填充参数、价格、租赁信息
  popup.innerHTML = `
    <h2>${data.name}</h2>
    <div class="section">
      <h3>核心参数</h3>
      ${data.params.map(p => `<p>${p.name}: ${p.value}</p>`).join('')}
    </div>
    ...（价格和租赁信息类似填充）
  `;
  document.body.appendChild(popup);
}

油猴（Tampermonkey）插件-机械设备信息爬取与聚合产品方案
一、方案核心定位
1. 核心目标
打造一款轻量化、免开发、即装即用的油猴插件，聚焦工程机械（汽车起重机为主，可扩展至挖掘机、履带吊等），自动爬取并聚合「新机价格+二手机价格+租赁价格+核心型号参数」四大核心信息，无需跳转平台，不打断用户浏览流程，实现“选中关键词→一键触发→精准展示”的高效查询体验。
2. 目标用户
工程机械采购人员（新机批量采购、二手机低成本选型）
工程施工方（查询设备租赁价格，核算施工成本）
行业从业者（快速比对设备参数与价格，开展业务对接）
零基础用户（无需代码能力，复制脚本即可使用，操作门槛为0）
3. 核心优势
信息全覆盖：一站式整合“新机+二手+租赁+参数”，无需切换多个平台查询；
轻量化便捷：油猴插件原生适配所有浏览器，无软件安装门槛，不占用本地存储空间；
精准高效：支持模糊/精准关键词匹配，结果悬浮展示，无需跳转页面，提升查询效率；
安全合规：仅爬取公开可访问的行业信息，纯前端请求，无账号封号风险，数据可交叉验证；
可自定义：支持用户手动补充设备型号、更新价格区间，适配个性化需求。

function searchEquipment(keyword) {
  // 本地缓存优先
  const cachedData = GM_getValue(keyword);
  if (cachedData) {
    showResult(cachedData);
    return;
  }

  // 多线程并发获取数据
  Promise.all([
    getParamsFromLmjx(keyword),   // 中国路面机械网参数
    getPriceFromXcmg(keyword),    // 徐工官网价格
    getUsedPriceFromTieba(keyword),// 铁甲二手机价格
    getRentPriceFromTieba(keyword)// 铁甲租赁价格
  ]).then(showResult);
}

// 3. 结果展示
function showResult(data) {
  // 创建悬浮弹窗

  popup.id = 'machinery-info-popup';

      ${data.params.map(p => `<p>${p.name}: ${p.value}</p>`).join('')}
    </div>
